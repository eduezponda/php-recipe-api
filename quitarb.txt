<meta charset="UTF-8">
<meta name="viewport" content="width=device-width, initial-scale=1.0">
<title>Cambiar texto HTML</title>
</head>
<body>

<div id="miDiv">
    Esto es un texto de ejemplo <b>en negrita</b> y <b>otro texto en negrita</b>.
</div>

<script>
    // Selecciona todos los elementos <b> dentro de #miDiv
    var elementosNegrita = document.querySelectorAll('#miDiv b');
    
    // Itera sobre cada elemento <b>
    elementosNegrita.forEach(function(elemento) {
        // Reemplaza el texto dentro del elemento <b> con su texto hijo (texto normal)
        elemento.parentNode.replaceChild(document.createTextNode(elemento.textContent), elemento);
    });
</script>

</body>
